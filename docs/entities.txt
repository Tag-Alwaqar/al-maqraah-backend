User extends BaseEntity:
    code: string //  generated
    password: string
    name: string
    phone: string
    gender: boolean
    born_at: Date // (2000-10-10)
    approved: boolean
    forget_pass_token?: string
    admin?: Admin
    teacher?: Teacher
    student?: Student

Admin:
    is_super: boolean
    user: User
    groups: Group[]
    fees: Fees[]

Teacher:
    user: User
    quraan_evaluations: QuraanEvaluation[]
    sharia_evaluations: ShariaEvaluation[]

Student:
    user: User
    current_surah: number => 1 => 114
    current_ayah?: number
    groups: Group[]
    fees: Fees[]
    quraan_evaluations: QuraanEvaluation[]
    sharia_evaluations: ShariaEvaluation[]

Fees:
    admin_id: number
    group_id: number
    student_id: number
    price: number // in EGP

GroupEnum:
    Quraan = quraan
    Sharia = sharia

Group extends BaseEntity:
    admin_id: number
    name: string
    type: GroupEnum
    gender: boolean
    students: Student[]
    quraan_evaluations: QuraanEvaluation[]
    sharia_evaluations: ShariaEvaluation[]

Revision:
    start_surah: number => 1 => 114
    end_surah: number => 1 => 114
    grade?: number => 0 => 5

NewSurah:
    start_ayah: number
    end_ayah: number
    grade?: number => 0 => 5

Evaluation:
    taken_at: Date and time
    group_id: number
    student_id: number
    teacher_id: number
    ethics_grade: boolean
    duration: number // in minutes

QuraanEvaluation extends Evaluation:
    current_revision: Revision
    next_revision: Revision
    current_new_surah: NewSurah
    next_new_surah: NewSurah
    tajweed_grade: number => 0 => 5

ShariaEvaluation extends Evaluation:
    attended: boolean

ExamEvaluation extends BaseEntity:
    name: string
    group_id: number
    student_id: number
    group_type: GroupEnum
    final_score: number
    grade: number
